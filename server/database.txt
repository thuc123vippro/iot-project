CREATE DATABASE IF NOT EXISTS smart_home_db;
USE smart_home_db;
-- Tạo bảng sensors (Cảm biến)
CREATE TABLE sensors (
    sensor_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    unit VARCHAR(50),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Tạo bảng sensor_data (Dữ liệu cảm biến gửi về)
CREATE TABLE sensor_data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    sensor_id INT NOT NULL,
    value FLOAT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_sensor_data_sensors 
        FOREIGN KEY (sensor_id) REFERENCES sensors(sensor_id) 
        ON DELETE CASCADE
);

-- Tạo bảng devices (Thiết bị điều khiển - Đèn, quạt, v.v.)
CREATE TABLE devices (
    device_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    current_status ENUM('on', 'off', 'waiting') DEFAULT 'off',
    gpio_pin INT,
    mqtt_topic VARCHAR(255),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Tạo bảng action_history (Lịch sử bật/tắt thiết bị)
CREATE TABLE action_history (
    id INT AUTO_INCREMENT PRIMARY KEY,
    device_id INT NOT NULL,
    action VARCHAR(255),
    status ENUM('on', 'off', 'waiting'),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_action_history_devices 
        FOREIGN KEY (device_id) REFERENCES devices(device_id) 
        ON DELETE CASCADE
);
